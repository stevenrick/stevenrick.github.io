{{ index $sections 0 | strings.TrimSpace }}

{{- end -}}
{{- /* Process sections as cards */ -}} {{- range $index, $section := after 1 $sections -}} {{- if or (hasPrefix $section " Current") (hasPrefix $section " Prior") -}}

{{ strings.TrimSpace (index (split $section "\n") 0) | strings.TrimPrefix " " }}

{{- $lines := after 1 (split $section "\n") -}} {{- $body := delimit $lines "\n" | strings.TrimSpace -}} {{ $body | markdownify }}
{{- end -}} {{- end -}} {{- /* CTA section */ -}} {{- if gt (len $mainAndCta) 1 -}} {{- $ctaContent := index $mainAndCta 1 | strings.TrimSpace -}} {{- if ne $ctaContent "" -}}
{{ $ctaContent | markdownify }}
{{- end -}} {{- end -}}
{{ end }}
